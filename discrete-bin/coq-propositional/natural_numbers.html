Fixpoint first_n_aux (n:nat)(m:nat) :=
  match n with
    0 => nil
    | S p => m :: first_n_aux p (m+1)
  end.

Eval compute in first_n_aux 2 0.

Definition first_n (n:nat) := first_n_aux n 0.

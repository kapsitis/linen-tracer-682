Lemma Jan1: forall a b: Prop, ~~((a \/ (b <-> a)) \/ b).
Proof.
  intros a b.
  assert ((a \/ (b <-> a)) \/ b) as H.
  pose (classic b) as H2.
  destruct H2 as [bTrue | bFalse].
  right.
  exact bTrue.
  left.
  pose (classic a) as H3.
  destruct H3 as [aTrue | aFalse].
  left.
  exact aTrue.
  right.
  split.
  intros bTrue.
  contradiction (bFalse bTrue).
  intros aTrue.
  contradiction (aFalse aTrue).
  unfold not.
  intros H4.
  contradiction (H4 H).
Qed.

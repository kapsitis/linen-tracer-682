Simple Tautologies

intros x. (For variables)
intros H. (For implications)
unfold not. (Replace "~")
unfold iff. (Replace "<->")
assert (...) as H1.
destruct H1 as [H2 H3].
destruct H1 as [H2 | H3].
exact H.
apply H.

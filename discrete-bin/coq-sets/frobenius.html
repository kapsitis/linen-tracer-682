Theorem frobenius (A: Set) (p: A -> Prop) (q : Prop) :
  (exists x: A, q /\ p x) <-> (q /\ exists x: A, p x).
Proof.
  split.
  intros.

  split.
  assumption.
  exists y.
  assumption.
  intros.
  destruct H as [H1 [y H2]].
  exists y.
  split.
  assumption.
  assumption.
Qed.

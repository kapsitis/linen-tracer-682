  0 => 0
  | S k => (SeqA k) + (k + 1) * (3 * (k + 1) + 1)
end.

Require Import Arith.

Lemma Problem1 : forall n, SeqA n = n * (n + 1) * (n + 1).
Proof.
  induction n.
  simpl.
  reflexivity.
  assert (CubicIdentity : (S n) * (S n + 1) * (S n + 1) =
      n*(n+1)*(n+1) + 3*n*n + 7*n + 4).
  ring.
  rewrite CubicIdentity.
  clear CubicIdentity.
  rewrite <- IHn.
  simpl.
  ring.
Qed.
